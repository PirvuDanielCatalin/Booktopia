# Booktopia
<VirtualHost 127.0.0.3:80>
    DocumentRoot "D:/Proiecte/Booktopia/public"
    ServerName shop.booktopia
    # http://shop.booktopia:80/

    <Directory "D:/Proiecte/Booktopia/public">
        Options All
        AllowOverride All
        Require all granted
    </Directory>
</VirtualHost>
